#!/bin/bash
#this script locks/unlockes user while keeping its password and output to /var/log/karimuserlog


read -p "please enter a username=" username

x="$(passwd -S $username)"

if [[ $x == *"LK"* ]]

then
echo  "$username is initially Locked & Now unlocked"
usermod -U $username
echo "$(passwd -S $username) Now unlocked" >> /var/log/karimuserlog
else
echo  "$username is initially unlocked & Now locked"
usermod -L $username
echo "$(passwd -S $username) Now locked" >> /var/log/karimuserlog
fi

#end
